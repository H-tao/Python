from functools import wraps


def singleton(cls):
    @wraps(cls)
    def wrapper(*args, **kwargs):
        if not wrapper.obj:
            wrapper.obj = cls(*args, **kwargs)
        return wrapper.obj
    wrapper.obj = None
    return wrapper

@singleton
class A:
    pass

a1 = A()
a2 = A()
print(a1, a2)
print(a1 is a2)
