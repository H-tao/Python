```python
"""
限制一个类的Name属性，让其不能为空，否则报错
"""
class NameCannotBeEmpty:
    def __init__(self, name=None):
        if name is not None:
            self.name = name
        elif not getattr(self, 'name', None):
            raise ValueError("%s must have a name" % type(self).__name__)

a = NameCannotBeEmpty()
```
