```python
g_total = 0
g_name = "Apple"

class Fruit(object):
    total = 0

class Apple(Fruit):
    total = 0
    def __init__(self):
        self.name = "Apple"
        self.num_list = []
        self.score_dict = {}

class Banana(Fruit):
    def __init__(self):
        self.name = "Banana"
        self.num_list = [1, 2]
        self.score_dict = {"Math": 20, "English": 87}

class Cherry(Fruit):
    total = 99
    name = "Cherry Class Name"
    def __init__(self):
        self.name = "Cherry Instance Name"

f1, f2 = Fruit(), Fruit()
a1, a2 = Apple(), Apple()
b1, b2 = Banana(), Banana()
c1, c2 = Cherry(), Cherry()

# 相同的不可变对象
print(g_total is f1.total)
print(g_name is a1.name)
print(f1.total is f2.total)
print(a1.total is f1.total)
print(b1.total is f1.total)
print(c1.total is f1.total)



print(id(a1.name), id(a2.name), a1.name is a2.name)
print(id(a1.num_list), id(a2.num_list), a1.num_list is a2.num_list)
print(id(a1.score_dict), id(a2.score_dict), a1.score_dict is a2.score_dict)
```
